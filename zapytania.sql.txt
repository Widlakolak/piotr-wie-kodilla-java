SELECT I.ID, I.TITLE, IL.NAME AS LIST_NAME
FROM ISSUESS I
JOIN ISSUESLISTS IL ON I.LIST_ID = IL.ID;

SELECT I.ID, I.TITLE, U.FIRSTNAME, U.LASTNAME
FROM ISSUESS I
JOIN USERS U ON I.USER_ID = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(I.ID) AS TASK_COUNT
FROM USERS U
JOIN ISSUESS I ON U.ID = I.USER_ID
GROUP BY U.ID;
